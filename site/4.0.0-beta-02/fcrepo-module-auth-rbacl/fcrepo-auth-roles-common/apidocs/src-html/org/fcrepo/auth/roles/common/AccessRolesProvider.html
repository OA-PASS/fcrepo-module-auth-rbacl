049/**
050 * Provides the effective access roles for authorization.
051 *
052 * @author Gregory Jansen
053 */
054@Component
055public class AccessRolesProvider {
056
057    private static final Logger LOGGER = getLogger(AccessRolesProvider.class);
058
059    public static final Map<String, List<String>> DEFAULT_ACCESS_ROLES = emptyMap();
060
061    /**
062     * Get the roles assigned to this Node. Optionally search up the tree for
063     * the effective roles.
064     *
065     * @param node the subject Node
066     * @param effective if true then search for effective roles
067     * @return a set of roles for each principal
068     */
069    public Map<String, List<String>> getRoles(final Node node, final boolean effective) throws RepositoryException {
070        final Map<String, List<String>> data = new HashMap<>();
071        final Session session = node.getSession();
072        registerPrefixes(session);
073        if (node.isNodeType(rbaclAssignable.getQualified())) {
074            getAssignments(node, data);
075            return data;
076        }
077        if (effective) { // look up the tree
078            try {
079                for (Node n = node.getParent(); n != null; n = n.getParent()) {
080                    if (n.isNodeType(rbaclAssignable.getQualified())) {
081                        if (LOGGER.isDebugEnabled()) {
082                            LOGGER.debug("effective roles are assigned at node: {}", n.getPath());
083                        }
084                        getAssignments(n, data);
085                        if (LOGGER.isDebugEnabled()) {
086                            for (final Map.Entry<String, List<String>> entry : data.entrySet()) {
087                                LOGGER.debug("{} has role(s) {}", entry.getKey(), entry.getValue());
088                            }
089                        }
090                        return data;
091                    }
092                }
093            } catch (final ItemNotFoundException e) {
094                LOGGER.trace("Subject not found, using default access roles", e);
095                return DEFAULT_ACCESS_ROLES;
096            }
097        }
098        return null;
099    }
100
101    /**
102     * @param node
103     * @param data
104     * @throws RepositoryException
105     */
106    private void getAssignments(final Node node, final Map<String, List<String>> data)
107        throws RepositoryException {
108
109        if (node.isNodeType(rbaclAssignable.getQualified())) {
110            try {
111                final Node rbacl = node.getNode(JcrName.rbacl.getQualified());
112                LOGGER.debug("got rbacl: {}", rbacl);
113                for (final NodeIterator ni = rbacl.getNodes(); ni.hasNext();) {
114                    final Node assign = ni.nextNode();
115                    final String principalName =
116                            assign.getProperty(principal.getQualified())
117                                    .getString();
118                    if (principalName == null ||
119                            principalName.trim().length() == 0) {
120                        LOGGER.warn("found empty principal name on node {}",
121                                    node.getPath());
122                    } else {
123                        List<String> roles = data.get(principalName);
124                        if (roles == null) {
125                            roles = new ArrayList<>();
126                            data.put(principalName, roles);
127                        }
128                        for (final Value v : assign.getProperty(
129                                role.getQualified()).getValues()) {
130                            if (v == null || v.toString().trim().length() == 0) {
131                                LOGGER.warn("found empty role name on node {}",
132                                            node.getPath());
133                            } else {
134                                roles.add(v.toString());
135                            }
136                        }
137                    }
138                }
139            } catch (final PathNotFoundException e) {
140                LOGGER.info(
141                             "Found rbaclAssignable mixin without a corresponding node at {}",
142                             node.getPath());
143            }
144        }
145    }
146
147    /**
148     * Assigns the given set of roles to each principal.
149     *
150     * @param node the Node to edit
151     * @param data the roles to assign
152     */
153    public void postRoles(final Node node, final Map<String, Set<String>> data)
154        throws RepositoryException {
155        final Session session = node.getSession();
156        registerPrefixes(session);
157        if (!node.isNodeType(rbaclAssignable.getQualified())) {
158            node.addMixin(rbaclAssignable.getQualified());
159            LOGGER.debug("added rbaclAssignable type");
160        }
161
162        Node acl;
163
164        if (node.hasNode(rbacl.getQualified())) {
165            acl = node.getNode(rbacl.getQualified());
166            for (final NodeIterator ni = acl.getNodes(); ni.hasNext();) {
167                ni.nextNode().remove();
168            }
169        } else {
170            acl = node.addNode(rbacl.getQualified(), Rbacl.getQualified());
171        }
172
173        for (final Map.Entry<String, Set<String>> entry : data.entrySet()) {
174            final Node assign = acl.addNode(assignment.getQualified(), Assignment.getQualified());
175            assign.setProperty(principal.getQualified(), entry.getKey());
176            assign.setProperty(role.getQualified(), toArray(entry.getValue(), String.class));
177        }
178    }
179
180    /**
181     * Deletes all roles assigned on this node and removes the mixin type.
182     *
183     * @param node
184     */
185    public void deleteRoles(final Node node) throws RepositoryException {
186        final Session session = node.getSession();
187        registerPrefixes(session);
188        if (node.isNodeType(rbaclAssignable.getQualified())) {
189            // remove rbacl child
190            try {
191                final Node rbacl = node.getNode(JcrName.rbacl.getQualified());
192                rbacl.remove();
193            } catch (final PathNotFoundException e) {
194                LOGGER.debug("Cannot find node: {}", node, e);
195            }
196            // remove mixin
197            node.removeMixin(rbaclAssignable.getQualified());
198        }
199    }
200
201    /**
202     * Finds effective roles assigned to a path, using first real ancestor node.
203     *
204     * @param absPath the real or potential node path
205     * @return the roles assigned to each principal
206     * @throws RepositoryException
207     */
208    public Map<String, List<String>> findRolesForPath(final Path absPath,
209            final Session session) throws RepositoryException {
210        Node node = null;
211        for (Path p = absPath; p != null; p = p.getParent()) {
212            try {
213                if (p.isRoot()) {
214                    node = session.getRootNode();
215                } else {
216                    node = session.getNode(p.getString());
217                }
218                break;
219            } catch (final PathNotFoundException e) {
220                LOGGER.debug("Cannot find node: {}", p, e);
221            }
222        }
223        return this.getRoles(node, true);
224    }
