16  package org.fcrepo.auth.roles.basic;
17  
18  import java.util.Set;
19  
20  import javax.jcr.Session;
21  
22  import org.fcrepo.auth.roles.common.AbstractRolesAuthorizationDelegate;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  /**
27   * @author Gregory Jansen
28   */
29  public class BasicRolesAuthorizationDelegate extends AbstractRolesAuthorizationDelegate {
30  
31      private static final Logger LOGGER = LoggerFactory
32              .getLogger(BasicRolesAuthorizationDelegate.class);
33  
34      /*
35       * (non-Javadoc)
36       * @see org.fcrepo.auth.roles.AbstractRolesAuthorizationDelegate#rolesHavePermission(final
37       * String absPath, final String[] actions, final Set<String> roles)
38       */
39      @Override
40      public boolean rolesHavePermission(final Session userSession,
41              final String absPath,
42              final String[] actions, final Set<String> roles) {
43          if (roles.isEmpty()) {
44              LOGGER.debug("A caller without content roles can do nothing in the repository.");
45              return false;
46          }
47          if (roles.contains("admin")) {
48              LOGGER.debug("Granting an admin role permission to perform any action.");
49              return true;
50          }
51          if (roles.contains("writer")) {
52              if (absPath.contains(AUTHZ_DETECTION)) {
53                  if (actions.length == 1 && "read".equals(actions[0])) {
54                      LOGGER.debug("Granting reader role permission to perform a read action.");
55                      return true;
56                  } else {
57                      LOGGER.debug("Denying writer role permission to perform an action on an ACL node.");
58                      return false;
59                  }
60              } else {
61                  LOGGER.debug("Granting writer role permission to perform any action on a non-ACL node.");
62                  return true;
63              }
64          }
65          if (roles.contains("reader")) {
66              if (actions.length == 1 && "read".equals(actions[0])) {
67                  LOGGER.debug("Granting reader role permission to perform a read action.");
68                  return true;
69              }
70              LOGGER.debug("Denying reader role permission to perform a non-read action.");
71              return false;
72          }
73          LOGGER.error("There are roles in session that aren't recognized by this authorization delegate: {}",
74                       roles);
75          return false;
76      }
77  
78  }
