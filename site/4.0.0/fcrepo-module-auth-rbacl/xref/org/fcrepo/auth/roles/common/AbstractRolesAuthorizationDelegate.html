38  /**
39   * Policy enforcement point for roles-based authentication
40   * @author Gregory Jansen
41   */
42  public abstract class AbstractRolesAuthorizationDelegate implements FedoraAuthorizationDelegate {
43  
44      private static final Logger LOGGER = LoggerFactory
45              .getLogger(AbstractRolesAuthorizationDelegate.class);
46  
47      protected static final String AUTHZ_DETECTION = "/{" +
48              Constants.JcrName.NS_URI + "}";
49  
50      private static final String[] REMOVE_ACTIONS = {"remove"};
51  
52      @Autowired
53      private AccessRolesProvider accessRolesProvider = null;
54  
55      @Autowired
56      private SessionFactory sessionFactory = null;

58      /**
59       * Gather effectives roles
60       *
61       * @param principals effective principals
62       * @return set of effective content roles
63       */
64      public static Set<String>
65      resolveUserRoles(final Map<String, List<String>> acl,
66                      final Set<Principal> principals) {
67          final Set<String> roles = new HashSet<>();
68          for (final Principal p : principals) {
69              final List<String> matchedRoles = acl.get(p.getName());
70              if (matchedRoles != null) {
71                  LOGGER.debug("request principal matched role assignment: {}", p.getName());
72                  roles.addAll(matchedRoles);
73              }
74          }
75          return roles;
76      }

78      @Override
79      public boolean hasPermission(final Session session, final Path absPath, final String[] actions) {
80          LOGGER.debug("Does user have permission for actions: {}, on path: {}", actions, absPath);
81          final boolean permission = doHasPermission(session, absPath, actions);
82  
83          LOGGER.debug("Permission for actions: {}, on: {} = {}", actions, absPath, permission);
84          return permission;
85      }
86  
87      private boolean doHasPermission(final Session session, final Path absPath, final String[] actions) {
88          final Set<String> roles;
89  
90          final Principal userPrincipal = getUserPrincipal(session);
91          if (userPrincipal == null) {
92              return false;
93          }
94  
95          final Set<Principal> allPrincipals = getPrincipals(session);
96          if (allPrincipals == null) {
97              return false;
98          }
99  
100         try {
101             final Session internalSession = sessionFactory.getInternalSession();
102             final Map<String, List<String>> acl =
103                     accessRolesProvider.findRolesForPath(absPath,
104                             internalSession);
105             roles = resolveUserRoles(acl, allPrincipals);
106             LOGGER.debug("roles for this request: {}", roles);
107         } catch (final RepositoryException e) {
108             throw new RepositoryRuntimeException("Cannot look up node information on " + absPath +
109                     " for permissions check.", e);
110         }
111 
112         if (LOGGER.isDebugEnabled()) {
113             LOGGER.debug("roles: {}, actions: {}, path: {}", roles, actions, absPath);
114             if (actions.length > 1) { // have yet to see more than one
115                 LOGGER.debug("FOUND MULTIPLE ACTIONS: {}", Arrays
116                         .toString(actions));
117             }
118         }
119 
120         if (actions.length == 1 && "remove_child_nodes".equals(actions[0])) {
121             // in roles-based ACLs, the permission to remove children is
122             // conferred by earlier check for "remove_node" on the child node
123             // itself.
124             return true;
125         }
126 
127         if (!rolesHavePermission(session, absPath.toString(), actions, roles)) {
128             return false;
129         }
130 
131         if (actions.length == 1 && "remove".equals(actions[0])) {
132             // you must be able to delete all the children
133             // TODO make recursive/ACL-query-based check configurable
134             return canRemoveChildrenRecursive(session, absPath.toString(),
135                     allPrincipals, roles);
136         }
137         return true;
138     }
139 
140     private static Principal getUserPrincipal(final Session session) {
141         final Object value = session.getAttribute(FEDORA_USER_PRINCIPAL);
142         if (value instanceof Principal) {
143             return (Principal) value;
144         }
145         return null;
146     }
147 
148     @SuppressWarnings("unchecked")
149     private static Set<Principal> getPrincipals(final Session session) {
150         final Object value = session.getAttribute(FEDORA_ALL_PRINCIPALS);
151         if (value instanceof Set<?>) {
152             return (Set<Principal>) value;
153         }
154         return null;
155     }

157     /**
158      * @param userSession the user session
159      * @param parentPath the parent path
160      * @param allPrincipals all principals
161      * @param parentRoles the roles on the parent
162      * @return true if permitted
163      */
164     private boolean canRemoveChildrenRecursive(final Session userSession,
165                                                final String parentPath,
166                                                final Set<Principal> allPrincipals,
167                                                final Set<String> parentRoles) {
168         try {
169             final Session internalSession = sessionFactory.getInternalSession();
170             LOGGER.debug("Recursive child remove permission checks for: {}",
171                          parentPath);
172             final Node parent = internalSession.getNode(parentPath);
173             if (!parent.hasNodes()) {
174                 return true;
175             }
176             final NodeIterator ni = parent.getNodes();
177             while (ni.hasNext()) {
178                 final Node n = ni.nextNode();
179                 // are there unique roles?
180                 final Set<String> roles;
181                 final Map<String, List<String>> acl = accessRolesProvider.getRoles(n, false);
182 
183                 if (acl != null) {
184                     roles = resolveUserRoles(acl, allPrincipals);
185                 } else {
186                     roles = parentRoles;
187                 }
188                 if (rolesHavePermission(userSession, n.getPath(),
189                         REMOVE_ACTIONS,
190                         roles)) {
191 
192                     if (!canRemoveChildrenRecursive(userSession, n.getPath(),
193                             allPrincipals, roles)) {
194                         return false;
195                     }
196                 } else {
197                     LOGGER.info("Remove permission denied at {} with roles {}", n.getPath(), roles);
198                     return false;
199                 }
200             }
201             return true;
202         } catch (final RepositoryException e) {
203             throw new RepositoryRuntimeException(
204                     "Cannot lookup child permission check information for " +
205                             parentPath, e);
206         }
207     }

209     /**
210      * Subclasses must override this method to determine permissions based on
211      * supplied roles.
212      * 
213      * @param userSession the user session
214      * @param absPath path to the object
215      * @param actions requested action
216      * @param roles effective roles for this request and content
217      * @return true if role has permission
218      */
219     public abstract boolean rolesHavePermission(final Session userSession, final String absPath,
220             final String[] actions, final Set<String> roles);
